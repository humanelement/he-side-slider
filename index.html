Human Element (HE) "Side Slider" Demo

A lightweight plugin to create horizontal or vertical content sliders.

Define a slider on your page with heSideSlider.init( ... );

Example 1

Kitty Carousel

/*JavasScript*/

jQuery(document).ready(function(){

    //example 1
    heSideSlider.init({
     wrap:'.example1:first', //selector: define the html wrapper
     slider:'.track:first', //selector: define the html element, moving "track"
     view:'.screen', //selector: define the html element that contains one or more of the "items" as child element(s)
     prev_btn:'.move-left:first', //selector: define previous button html element
     next_btn:'.move-right:first', //selector: define next button html element
     items_per_view:3 //how many items should appear per view?
    });

});
/*Custome SCSS*/

.example1{
    width:700px;
    &[data-slide-mode="slide"] .he_slider .he_view .he_item{width:33.3333% !important;padding:0 2px;}
}

Example 2

Meowy Mover

Set vertical, instead of horizontal, orientation.

/*JavasScript*/

jQuery(document).ready(function(){

    //example 2
    heSideSlider.init({
     wrap:'.example2:first',
     slider:'.track:first',
     view:'.screen',
     prev_btn:'.move-up:first',
     next_btn:'.move-down:first',
     items_per_view:3
     slide_mode:'slide-vertical' //vertical orientation
    });

});
/*Custome SCSS*/

.example2{
    width:146px;

    .slider-boundary{height:450px;}
    &[data-slide-mode="slide-vertical"] .he_slider .he_view .he_item{height:33.3333% !important;padding:2px 0;}
}

Example 3

Feline Fader

Create a fader, instead of slider.

/*JavasScript*/

jQuery(document).ready(function(){

    //example 3
    heSideSlider.init({
     wrap:'.example3:first',
     slider:'.track:first',
     view:'.screen',
     items_per_view:1,
     slide_mode:'fade', //fade transition
     time_between_autoslide:1000, //pause duration between slide transitions
     transition_speed:750, //fade effect speed
     circular_next_prev:true, //cycle back to the first slide after reaching the last slide
     swipe_slide:false //disable touch-screen ability to go to prev/next slide on swipe
    });

});

Example 4

Two Row Tabby

Feel free to style the layout within a "view" so that the "items" inside the "view" can wrap to multiple rows.

/*JavasScript*/

jQuery(document).ready(function(){

    //example 4
    heSideSlider.init({
     wrap:'.example4:first',
     slider:'.track:first',
     view:'.screen',
     prev_btn:'.move-left:first',
     next_btn:'.move-right:first',
     items_per_view:4
    });

});
/*Custome SCSS*/

.example4{
    width:700px;

    &[data-slide-mode="slide"] .he_view{white-space:normal;}

    &[data-slide-mode="slide"] .he_slider .he_view .he_item{width:50% !important;padding:2px;}
}

Example 5

Sneaky Switcheroo

Horizontal at wider window widths and vertical at thinner window widths...

/*JavasScript*/

jQuery(document).ready(function(){

    //example 4
    heSideSlider.init({
     wrap:'.example5:first',
     slider:'.track:first',
     view:'.screen',
     prev_btn:'.move-left:first',
     next_btn:'.move-right:first',
     items_per_view:4,
     at_or_below_widths:{ //apply different settings, depending on window width
        '800':{ //at 800px window width...
         items_per_view:3, //show 3 items per view
         slide_mode:'slide-vertical' //vertical orientation
        }
     }
    });

});
/*Custome SCSS*/

.example5{

    &[data-slide-mode="slide"]{
        width:850px;

         .he_slider .he_view .he_item{width:25% !important;padding:0 2px;}
    }
    &[data-slide-mode="slide-vertical"]{
        width:146px;

        .slider-boundary{height:450px;}

        .he_slider .he_view .he_item{height:33.3333% !important;padding:2px 0;}
    }
}
